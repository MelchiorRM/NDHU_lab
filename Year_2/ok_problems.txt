#include <bits/stdc++.h>
using namespace std;

int main() {
    int N;
    cin >> N;

    int total = 1 << N;  // 2^N

    for (int i = 0; i < total; i++) {
        int gray = i ^ (i >> 1);  // academic formula

        // print gray code in N bits
        for (int b = N - 1; b >= 0; b--) {
            cout << ((gray >> b) & 1);
        }
        cout << "\n";
    }

    return 0;
}

#include <iostream>
#include <vector>
#include <string>
#include <algorithm> // for std::reverse
// can replace with #include <bits/stdc++.h> if allowed
using namespace std;

// Recursive function to generate Gray Codes
vector<string> generateGrayCode(int n) {
    // Base Case: N=1
    if (n == 1) {
        return {"0", "1"};
    }

    // Recursive Step: Get the sequence for N-1
    vector<string> prevGray = generateGrayCode(n - 1);
    vector<string> currentGray;

    // 1. Left Half (Order): Add '0' prefix
    for (int i = 0; i < prevGray.size(); i++) {
        currentGray.push_back("0" + prevGray[i]);
    }

    // 2. Right Half (Mirror/Reverse): Add '1' prefix
    // We iterate backwards through the previous list
    for (int i = prevGray.size() - 1; i >= 0; i--) {
        currentGray.push_back("1" + prevGray[i]);
    }

    return currentGray;
}

int main() {
    int N;
    if (cin >> N) {
        vector<string> result = generateGrayCode(N);
        
        // Output the result
        for (const string &s : result) {
            cout << s << endl;
        }
    }
    return 0;
}

#include <iostream>
#include <vector>
#include <map>

using namespace std;

// A structure to hold Binary Tree pointers
struct Node {
    int left = 0;  // Represents the First Child in Original Tree
    int right = 0; // Represents the Next Sibling in Original Tree
};

int main() {
    int M;
    if (!(cin >> M)) return 0;

    // Using a map to handle non-contiguous or large IDs easily.
    // If IDs are small (e.g., < 1000), you could use Node tree[1001];
    map<int, Node> tree;

    int parent, child;
    char type;

    // 1. Build the Binary Tree from input
    for (int i = 0; i < M; i++) {
        cin >> parent >> child >> type;
        if (type == 'L') {
            tree[parent].left = child;
        } else if (type == 'R') {
            tree[parent].right = child;
        }
    }

    int targetID;
    cin >> targetID;

    // 2. Find children in the Original General Tree
    // Step A: Check if the target has a Left Child (First Child)
    if (tree[targetID].left == 0) {
        cout << "No children" << endl;
    } else {
        // Step B: Start with the First Child
        int current = tree[targetID].left;
        bool firstPrint = true;

        // Step C: Traverse the "Right Chain" to find all siblings
        while (current != 0) {
            if (!firstPrint) cout << " ";
            cout << current;
            firstPrint = false;
            
            // Move to the next sibling (which is the Right child in Binary representation)
            current = tree[current].right; 
        }
        cout << endl;
    }

    return 0;
}

#include <bits/stdc++.h>
using namespace std;

int main() {
    int M;
    cin >> M;

    // We assume nodes are up to maybe 10^5 or small; use maps just in case
    unordered_map<int, int> leftChild;
    unordered_map<int, int> rightChild;

    for (int i = 0; i < M; i++) {
        int parent, child;
        char type;
        cin >> parent >> child >> type;

        if (type == 'L')
            leftChild[parent] = child;
        else
            rightChild[parent] = child;
    }

    int target;
    cin >> target;

    // Start with left child
    if (leftChild.find(target) == leftChild.end()) {
        cout << "No children";
        return 0;
    }

    int cur = leftChild[target];
    vector<int> children;

    // Traverse right-chain
    while (cur != 0) {
        children.push_back(cur);

        // Move to sibling
        if (rightChild.find(cur) != rightChild.end())
            cur = rightChild[cur];
        else
            break;
    }

    // Output
    for (int c : children)
        cout << c << " ";

    return 0;
}
